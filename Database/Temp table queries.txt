--search result query
select
  g.Title,
  f.Name,
  d.CompanyName,
  g.price
into #game_search --temp table
from
  Game g,
  Genre f,
  Developer d
where
  g.idGenres = f.idGenres
  and d.idDevelopers = g.idDevelopers
  and g.name = TextBox


--Developer profile team list box query
select
  Employee.name,
  Employee.designation
into #team_list --temp table
from
  Employee,
  Developer
where
  Employee.idDeveloper = Developer.idDeveloper

--Developer profile game list box
select
  Game.title
into *games_list--temp table
from
  Game,
  Developer
where
  Game.idDeveloper = Developer.idDeveloper
